package com.example.vodamanewakala.db

import androidx.room.ColumnInfo
import androidx.room.Entity
import androidx.room.PrimaryKey

@Entity(tableName = "balance_table")
data class Balance(

    @PrimaryKey(autoGenerate = true)
    @ColumnInfo(name="balanceid")
    val balanceid: Int,
    @ColumnInfo(name="transid")
    val transid: String,
    @ColumnInfo(name="balanceinout")
    val balanceinout: String,
    @ColumnInfo(name="balance")
    val balance: String,
    @ColumnInfo(name="status")
    val status: Int,
    @ColumnInfo(name="wakalaorder")
    val wakalaorder: String,
)
