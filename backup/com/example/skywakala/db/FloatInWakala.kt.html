package com.example.vodamanewakala.db

import androidx.room.ColumnInfo

data class FloatInWakala(
        @ColumnInfo(name="floatinid")
        val floatinid: Int,
        @ColumnInfo(name="transid")
        val transid: String,
        @ColumnInfo(name="amount")
        val amount: String,
        @ColumnInfo(name="balance")
        val balance: String,
        @ColumnInfo(name="wakalaorder")
        val wakalaorder: String,
        @ColumnInfo(name="wakalaidkey")
        val wakalaidkey: String,
        @ColumnInfo(name="wakalaid")
        val wakalaid: String,
        @ColumnInfo(name="tigopesa")
        val tigopesa: String,
        @ColumnInfo(name="mpesa")
        val mpesa: String,
        @ColumnInfo(name="halopesa")
        val halopesa: String,
        @ColumnInfo(name="tpesa")
        val tpesa: String,
        @ColumnInfo(name = "airtelmoney")
        val airtelmoney: String,
        @ColumnInfo(name="tigoname")
        val tigoname: String,
        @ColumnInfo(name="vodaname")
        val vodaname: String,
        @ColumnInfo(name="haloname")
        val haloname: String,
        @ColumnInfo(name="ttclname")
        val ttclname: String,
        @ColumnInfo(name = "airtelname")
        val airtelname: String,
)
