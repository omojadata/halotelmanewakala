package com.example.vodamanewakala.viewmodel



import androidx.databinding.Bindable
import androidx.databinding.Observable
import androidx.lifecycle.MutableLiveData
import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.example.vodamanewakala.db.MobileRepository
import com.example.vodamanewakala.db.WakalaMkuu
import kotlinx.coroutines.Job
import kotlinx.coroutines.launch

class WakalaMkuuViewModel(private val repository: AirtelRepository):ViewModel(), Observable {

    val wakalaMkuu=repository.wakalaMkuu


    @Bindable
    val inputpass=MutableLiveData<String>()

    @Bindable
    val inputtigopesa=MutableLiveData<String>()

    @Bindable
    val inputmpesa=MutableLiveData<String>()

    @Bindable
    val inputhalopesa=MutableLiveData<String>()

    @Bindable
    val inputtpesa=MutableLiveData<String>()

    @Bindable
    val saveOrUpdateButtonText =MutableLiveData<String>()


    init {
        saveOrUpdateButtonText.value= "Save"

    }


    fun saveOrUpdate(){
        if (inputpass.value=="2390" && inputhalopesa!=null && inputmpesa!=null && inputtigopesa!=null &&
                inputtpesa.value!=null){
            val halopesa=inputhalopesa.value!!
            val tigopesa= inputtigopesa.value!!
            val mpesa=inputmpesa.value!!
            val tpesa =inputtpesa.value!!
            insert(WakalaMkuu(0 ,tigopesa,mpesa, halopesa, tpesa,1))
        }
        inputhalopesa.value=null
        inputtigopesa.value=null
        inputmpesa.value=null
        inputtpesa.value=null

    }
    fun insert(wakalaMkuu: WakalaMkuu): Job =
        viewModelScope.launch {
            repository.insertWakalaMkuu(wakalaMkuu)
        }

   

    override fun addOnPropertyChangedCallback(callback: Observable.OnPropertyChangedCallback?) {

    }

    override fun removeOnPropertyChangedCallback(callback: Observable.OnPropertyChangedCallback?) {

    }
}
